vector<int> parseInts(string str) {
    vector<int> v;
    stringstream ss(str);
    int val;
    char ch;
    while(ss>>val){
        v.push_back(val);
        ss>>ch; }
    return v;
}

vector<int> parseInts(string str) {
    vector<int> v;
    stringstream ss(str);
    string val;
    while(getline(ss,val,',')){
        v.push_back(stoi(val)); }
    return v;
}
